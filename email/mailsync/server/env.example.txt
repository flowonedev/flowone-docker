# Mail Sync Server Configuration
# Copy this file to .env and update values

# WebSocket Server
MAILSYNC_WS_HOST=0.0.0.0
MAILSYNC_WS_PORT=1235

# SSL (set to true if not behind reverse proxy)
MAILSYNC_SSL_ENABLED=false
MAILSYNC_SSL_KEY=/etc/letsencrypt/live/email.devcon1.hu/privkey.pem
MAILSYNC_SSL_CERT=/etc/letsencrypt/live/email.devcon1.hu/fullchain.pem

# JWT (must match PHP backend)
# RS256 (verify with a public key) is preferred; the server NEVER signs tokens.
# Point JWT_PUBLIC_KEY_PATH at the PEM public key, or drop jwt-public.pem in the
# server dir. If RS256 is set with no key the server logs FATAL and rejects all
# connections. JWT_SECRET is the HS256 fallback during migration.
JWT_PUBLIC_KEY_PATH=/path/to/jwt-public.pem
JWT_ALGORITHM=RS256
JWT_SECRET=your-jwt-secret-here

# IMAP Settings
IMAP_HOST=localhost
IMAP_PORT=993
IMAP_TLS=true
IMAP_VERIFY_CERT=false
IMAP_IDLE_TIMEOUT=1680000
IMAP_RECONNECT_DELAY=5000
IMAP_MAX_RECONNECT_ATTEMPTS=10

# Redis
REDIS_HOST=127.0.0.1
REDIS_PORT=6379
REDIS_PASSWORD=
REDIS_DATABASE=0

# --- Push notifications (this server sends ALL pushes) -----------------------
# The PHP backend only stores tokens + syncs them into Redis; this server reads
# them and delivers Web Push / FCM / APNs VoIP. Leave a block empty to disable
# that transport (the others still work).

# Web Push (VAPID). Must match the backend's VAPID_PUBLIC_KEY.
# Generate with: node ../../scripts/generate-vapid-keys.js
VAPID_PUBLIC_KEY=
VAPID_PRIVATE_KEY=
VAPID_SUBJECT=mailto:admin@yourdomain.com

# Firebase Cloud Messaging (native iOS/Android alert + Android call data push).
# Point at the Firebase Admin service account JSON. Unset/missing => FCM off.
FCM_SERVICE_ACCOUNT_PATH=/path/to/fcm-service-account.json
FCM_ENABLED=true

# APNs VoIP (PushKit) — rings the iOS Chat app's native CallKit screen.
# Token-based auth (.p8). The VoIP topic is APNS_VOIP_BUNDLE_ID + ".voip".
APNS_VOIP_ENABLED=true
APNS_VOIP_KEY_PATH=/path/to/apns-voip-key.p8
APNS_VOIP_KEY_ID=
APNS_VOIP_TEAM_ID=
APNS_VOIP_BUNDLE_ID=com.flowone.chat
# CRITICAL: must match the build's APNs environment, or VoIP pushes silently
# fail and incoming calls never ring on the phone:
#   - App Store / TestFlight builds  -> true  (api.push.apple.com)
#   - Xcode debug / dev (Sandbox)    -> false (api.sandbox.push.apple.com)
# Default when unset is true (production).
APNS_VOIP_PRODUCTION=true

# Event Store
EVENT_BUFFER_SIZE=100
EVENT_BUFFER_TTL=300

# Logging
LOG_LEVEL=info

# Performance
MAX_IDLE_CONNECTIONS=500
WS_HEARTBEAT_INTERVAL=30000
WS_CLIENT_TIMEOUT=60000
